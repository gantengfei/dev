<div class="webpagewrap"></div>

<script>
const host = window.location.host;
const WEB_PATH = host.includes('github') ? '/dev':''
const data = $.ajax({ async: false, url: `${WEB_PATH}/src/data/link.config.json` }).responseJSON;

let Htm = ``
data.links.forEach((item,index)=>{
  Htm+=`<h1 class="headtitle" id="${item.title}$" level="1">${item.title}</h1>`
  if(item.lists){
    Htm+=`<ul class="weblistwrap">`
    item.lists.forEach((mitem,mindex)=>{
      Htm+=`<li class="${mitem.isMark?'tag':''}"><a href="${mitem.href}" target="_blank"><span class="icon ${mitem.logoIcon}"></span><span>${mitem.title}</span><i>${mitem.brief}</i></a></li>`
    })
    Htm+=`</ul>`
  }
  if(item.children){
    item.children.forEach((citem,cindex)=>{
      Htm+=`<h2 class="headtitle" id="${citem.title}" level="2">${citem.title}</h2>`
      if(citem.lists){
        Htm+=`<ul class="weblistwrap">`
        citem.lists.forEach((scitem,scindex)=>{
          Htm+=`<li class="${scitem.isMark?'tag':''}"><a href="${scitem.href}" target="_blank"><span class="icon ${scitem.logoIcon}"></span><span>${scitem.title}</span><i>${scitem.brief}</i></a></li>`
        })
        Htm+=`</ul>`
      }
    })
  }
})

$('.webpagewrap').html(Htm)
</script>
