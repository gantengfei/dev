
1.**编辑代码片段**
``` html
<template>
  <div>AppContent</div>
</template>

<style lang="scss"></style>

<script lang="ts" setup>
import { onMounted, ref, reactive } from 'vue'

onMounted(() => {

})

</script>
```

``` html
<template>
  <div>AppContent</div>
</template>

<style lang="scss"></style>

<script lang="ts">
import { Options, Vue } from 'vue-class-component'
import { ref, reactive } from 'vue'

//@ts-ignore
@Options({
  name: "module-name",
  components: {}
})

export default class ModuleName extends Vue {

  mounted(): void {

  }

}
</script>
```

2.**打开网站：**[snippet generator](https://snippet-generator.app/)

`prefix` ：快捷键；`body` ：代码片段；`description` ：描述；
![输入图片说明](https://foruda.gitee.com/images/1706233587611829920/8f7c8dd0_4993153.png "")

3.**vscode打开选项：**

管理 → 用户代码片段（Configure User Snippets） → 快捷键应用在对应文件中，上面代码片段应用场景在vue中，选择`vue.json` → 将网站中生成的代码复制到json文件中。

![输入图片说明](https://foruda.gitee.com/images/1706233723473994249/b829c68f_4993153.png "")
![输入图片说明](https://foruda.gitee.com/images/1706233806103747748/cf5e5076_4993153.png "")
![输入图片说明](https://foruda.gitee.com/images/1706233967961364733/a4d76171_4993153.png "")

`vue.json`文件中配置
``` json
{
	"vue-snippet": {
		"prefix": "vue",
		"body": [
		  "<template>",
		  "  <div>AppContent</div>",
		  "</template>",
		  "",
		  "<style lang=\"scss\"></style>",
		  "",
		  "<script lang=\"ts\" setup>",
		  "import { onMounted, ref, reactive } from 'vue'",
		  "",
		  "onMounted(() => {",
		  "",
		  "})",
		  "",
		  "</script>"
		],
		"description": "vue-snippet"
	},
	"vue-class-snippet": {
		"prefix": "vue-class",
		"body": [
		  "<template>",
		  "  <div>AppContent</div>",
		  "</template>",
		  "",
		  "<style lang=\"scss\"></style>",
		  "",
		  "<script lang=\"ts\">",
		  "import { Options, Vue } from 'vue-class-component'",
		  "import { ref, reactive } from 'vue'",
		  "",
		  "//@ts-ignore",
		  "@Options({",
		  "  name: \"module-name\",",
		  "  components: {}",
		  "})",
		  "",
		  "export default class ModuleName extends Vue {",
		  "",
		  "  mounted(): void {",
		  "",
		  "  }",
		  "",
		  "}",
		  "</script>"
		],
		"description": "vue-class-snippet"
	}
}
```
